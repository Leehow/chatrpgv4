```python
from __future__ import annotations
import json
from pathlib import Path
from typing import Any

def load_role_templates(rules_json_dir: Path) -> dict[str, dict[str, Any]]:
    path = rules_json_dir / "npc-social-roles.json"
    data = json.loads(path.read_text())
    return {t["template_id"]: t for t in data.get("templates", [])}

def load_role_keywords(rules_json_dir: Path) -> dict[str, Any]:
    return json.loads((rules_json_dir / "npc-role-keywords.json").read_text()).get("mappings", {})

def expand_npc_social_roles(
    npcs_doc: dict[str, Any],
    templates: dict[str, dict[str, Any]],
    *,
    keywords: dict[str, Any] | None = None,
) -> dict[str, Any]:
    """确定性注入：按 NPC 的 relationship_to_investigators（结构化字段）映射到模板。
    已有 social_role 不覆盖；无匹配不注入。不扫自由文本。"""
    keywords = keywords or {}
    out_npcs = []
    for npc in (npcs_doc.get("npcs") or []):
        npc = dict(npc)
        if npc.get("social_role"):
            out_npcs.append(npc); continue
        rel = str(npc.get("relationship_to_investigators") or "").strip()
        mapping = keywords.get(rel) or {}
        template_id = mapping.get("template_id")
        template = templates.get(template_id) if template_id else None
        if template:
            role = {
                "authority_scope": list(template.get("authority_scope", [])),
                "responsibility_domains": list(template.get("responsibility_domains", [])),
                "chain_of_command": dict(template.get("chain_of_command", {"to_pc": "none", "to_group": "none"})),
                "duty_pressure": list(template.get("duty_pressure", [])),
                "initiative_style": mapping.get("initiative_style_override") or template.get("initiative_style", "consultative"),
                "delegation_policy": dict(template.get("delegation_policy", {"keeps": [], "delegates": []})),
            }
            npc["social_role"] = role
        out_npcs.append(npc)
    return {**npcs_doc, "npcs": out_npcs}

def expand_from_dir(npcs_doc: dict[str, Any], rules_json_dir: Path) -> dict[str, Any]:
    templates = load_role_templates(rules_json_dir)
    keywords = load_role_keywords(rules_json_dir)
    return expand_npc_social_roles(npcs_doc, templates, keywords=keywords)
```
